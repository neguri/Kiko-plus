## 2. 순열


``` cpp

#include <iostream>

int A[] = {1,2,3,4,5};

void swap(int a, int b)
{
  int t = arr[a];
  arr[a] = arr[b];
  arr[b] = t;
}

bool permutation(int n)
{
  int i = n-1;
  while( i>0 && A[i-1] >= A[i]) i--;
  
  if(i <= 0) return false;
  
  int j = n-1;
  while(A[i-1] >=  A[j] ) j--;
  swap(i-1,j);
  
  j = n-1;
  while(i<j) {
    swap(i++,j--);
  }
  
  return true;
}

int main()
{

  int n = sizeof(A)/sizeof(A[0]);
  
  do {
    for(int i = 0; i<n; i++) {
    	std::cout<<arr[i]<<' ';
    }std::cout<<std::endl;
  }while(permutation(n));
  return 0;
}
```
