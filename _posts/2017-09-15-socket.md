## 1. server

``` cpp
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <netinet/in.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <arpa/inet.h>
#include <fcntl.h>
#include <string.h>
 
 
#define BUF_LEN 128
 
int main(int argc, char* argv[])
{
    char buffer[BUF_LEN];
    struct sockaddr_in server_addr, client_addr;
    char temp[20];
    int server_fd, client_fd;
    int len, msg_size;
    char test[20];

    if( (server_fd = socket(AF_INET, SOCK_STREAM, 0)) ==  -1) {
        printf("can't open server\n");
        exit(0);
    }
 
    memset(&server_addr, 0x00, sizeof(server_addr));
 
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = htonl(INADDR_ANY);
    server_addr.sin_port = htons(atoi(argv[1]));
 
    if( bind(server_fd, (struct sockaddr*)&server_addr, sizeof(server_addr))<0){
      printf("server: can't bind\n");
      exit(0);
    }
 
    if( listen(server_fd, 5) < 0) {
      printf("server: can't bind\n");
      exit(0);
    }
 
 
    memset(buffer, 0x00, sizeof(buffer));
 
    len = sizeof(client_addr);
    printf(" port = %d\n", atoi(argv[1]));
    printf(" server -- waiting for connection\n");
 
    while(1) {
        printf(" server -- waiting for connection -- inside of while\n");
        client_fd = accept(server_fd, (struct sockaddr *)&client_addr, (socklen_t *)&len);
        printf("buffer len = %d\n", len);
        if(client_fd < 0) printf("server accept fail\n");
 
        inet_ntop(AF_INET, &client_addr.sin_addr.s_addr, temp, sizeof(temp));
        printf("server is connect : %s\n", temp);
 
        send(client_fd, "aaa aaa aaa", 0xff, 0);
        msg_size = recv(client_fd, (char *) buffer, BUF_LEN, 0);
        close(client_fd);
        printf("server : %s client closed\n", temp);
    }
 
    close(server_fd);
 
    return 0;
}
```

## 2. client


``` cpp
#include <stdio.h>
#include <unistd.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <string.h>
#include <arpa/inet.h>
#include <netinet/in.h>
#include <stdlib.h>
 
 
#define BUF_SIZE 1024
#define PORT 1234
#define IPADDR "127.0.0.1"
 
int main()
{
 
    int client_fd, len;
    struct sockaddr_in client_addr;
    char recv_data[1024];
 
    client_fd = socket(PF_INET, SOCK_STREAM, 0);
 
 
    client_addr.sin_addr.s_addr = inet_addr(IPADDR);
    client_addr.sin_family= AF_INET;
    client_addr.sin_port = htons(PORT);
 
    if( connect(client_fd, (struct sockaddr *)&client_addr, sizeof(client_addr)) ==  -1) {
        printf("connection error\n");
        close(client_fd);
        return -1;
    }
 
    recv(client_fd, recv_data, sizeof(recv_data), 0);
    printf("recv data = %s\n", recv_data);
 
    close(client_fd);
 
    return 0;
}
```
